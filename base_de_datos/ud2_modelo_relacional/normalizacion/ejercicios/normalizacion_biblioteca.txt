1FN - atributos atómicos y clave primaria definida
SOCIO (n_socio PK, nombre, apellido, dirección, telefono)
LIBRO (isbn PK, titulo, código_autor, nombre_autor, fecha_nacimiento_autor, 
fecha_muerte_autor, tipo_autor, paginas)
PRESTAMO (id PK, dias_prestamo, fecha_prestamo, fecha_esperada_devolucion, fecha_devolución
n_socio FK(SOCIO.n_socio), isbn FK(LIBRO.isbn))

2FN - atributos no clave dependen de toda la clave (si es compuesta)
SOCIO (n_socio PK, nombre, apellido, dirección, telefono)
LIBRO (isbn PK, titulo, paginas)
PRESTAMO (id PK, dias_prestamo, fecha_prestamo, fecha_esperada_devolucion, fecha_devolución
n_socio FK(SOCIO.n_socio), isbn FK(LIBRO.isbn))
AUTOR (cod_autor PK, nombre_autor, fecha_nacimiento_autor, fecha_muerte_autor)
ESCRIBE(isbn PK, codigo_autor PK, tipo_autor)

3FN - ningún atributo no clave depende de otro atributo no clave
SOCIO (n_socio PK, nombre, apellido, dirección, telefono)
LIBRO (isbn PK, titulo, paginas, dias_prestamo)
ESCRIBE (isbn PK, codigo_autor PK, tipo_autor)
AUTOR (cod_autor PK, nombre_autor, fecha_nacimiento_autor, fecha_muerte_autor)
PRESTAMO (id_prestamo PK, fecha_prestamo, fecha_esperada_devolucion, 
socio FK(SOCIO.n_socio), isbn FK(LIBRO.isbn_libro))
DEVOLUCION (isbn PK(LIBRO.isbn), fecha_prestamo PK, fecha_esperada_devolucion)
